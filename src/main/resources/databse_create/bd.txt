create table people
(
    id       integer generated by default as identity
        primary key,
    name     varchar not null,
    year     integer not null,
    username varchar
        unique,
    password varchar,
    role     varchar
);

create table book
(
    id          integer generated by default as identity
        primary key,
    name        varchar not null,
    author_name varchar not null,
    year        integer not null,
    id_person   integer
        constraint book_person_id_fkey
            references people
);
